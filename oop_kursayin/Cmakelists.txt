cmake_minimum_required(VERSION 3.18)
project(ash_compiler CXX)
 
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
 
add_executable(ashc
    main.cpp
    Lexer/lexer.cpp
    Parser/parser.cpp
    Sema/sema.cpp
    CodeGenerator/llvm_emitter.cpp
    Loader/loader.cpp
)
 
target_include_directories(ashc PRIVATE ${CMAKE_SOURCE_DIR})
target_compile_options(ashc PRIVATE -Wall -Wextra -O2)
